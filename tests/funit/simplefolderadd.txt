add a folder named dummyfolder at the Silva root
--------------------------------------------------

Create the browser object we'll be using.
    
    >>> from Products.Five.testbrowser import Browser
    >>> from Products.Silva.tests.SilvaTestCase import user_name, user_password
    >>> browser = Browser()
    >>> browser.addHeader('Authorization', 'Basic %s:%s' % (user_name, user_password))
    >>> browser.addHeader('Accept-Language', 'en-US')
    >>> browser.open(self.silva_url)
    >>> browser.getLink('click this link').click()
    >>> browser.url == self.silva_url + '/edit'
    True
    
Create folder:
   
    >>> browser.getControl(name='meta_type').value = ['Silva Folder', ]
    >>> browser.getControl(name='add_object:method').click()
    >>> browser.getControl(name='field_object_id').value = 'dummyfolder'
    >>> browser.getControl(name='field_object_title').value = 'Dummy Folder'
    >>> browser.getControl(name='add_submit').click()
    >>> "Dummy Folder" in browser.contents
    True


Recreate the same folder:

    >>> browser.getControl(name='meta_type').value = ['Silva Folder',]
    >>> browser.getControl(name='add_object:method').click()
    >>> browser.getControl(name='field_object_id').value = 'dummyfolder'
    >>> browser.getControl(name='field_object_title').value = 'Dummy Folder'
    >>> browser.getControl(name='add_submit').click()
    >>> "There is already an object with the id" in browser.contents
    True
